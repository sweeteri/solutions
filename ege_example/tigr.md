### Скрипт на все три задачи
```python
def f(step, limit, a): #ход, кол-во ходов, кол-во камней

    # если сумма куч больше победного значения, возвращаем победил ли целевой игрок
    # (четные ходы ваня, нечетные петя)
    if a > 130: return step % 2 == limit % 2

    if step == limit: return 0  # если программа дошла до этого момента, то это значит,
    # что победа не произошла за нужное количество ходов, поэтому возвращаем ложь.

    # необязательная часть: составляем список или кортеж со всевозможными ходами
    moves = f(step + 1, limit, a * 2), f(step + 1, limit, a + 2)

    # если четность хода целевого игрока совпадает с четностью количества всех ходов,
    # то возвращаем победу в хотя бы одном из случаев, иначе во всех.
    return any(moves) if (step + 1) % 2 == limit % 2 else all(moves)
    
    
for s in range(1, 121): # перебор кол-ва камней
    for lim in range(1, 5): # перебор кол-ва ходов
        if f(0, lim, s):
            print(s, lim)
            break # берем наименьшее значение ходов для победы 

```
